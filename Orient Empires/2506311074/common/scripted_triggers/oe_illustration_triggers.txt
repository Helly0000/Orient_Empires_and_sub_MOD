has_graphical_china_region_trigger = {
	trigger_if = {
		limit = {
			exists = capital_province
		}
		capital_province = {
			geographical_region = world_asia_china
		}
	}
	trigger_if = {
		limit = {
			exists = host.capital_province
			NOT = { exists = capital_province }
		}
		host.capital_province = {
			geographical_region = world_asia_china
		}
	}
	trigger_if = {
		limit = {
			exists = liege.capital_province
			NOT = { exists = capital_province }
		}
		liege.capital_province = {
			geographical_region = world_asia_china
		}
	}
	trigger_if = {
		limit = {
			exists = location.province_owner.capital_province
			NOT = { exists = capital_province }
		}
		location.province_owner.capital_province = {
			geographical_region = world_asia_china
		}
	}
}

artifact_use_chinese_building_visuals_trigger = {
	exists = culture
	culture = { has_building_gfx = chinese_building_gfx }
}

artifact_use_austronesian_unit_visuals_trigger = {
	culture = { has_unit_gfx = austronesian_unit_gfx }
}

artifact_use_indian_building_visuals_trigger = {
	exists = culture
	OR = {
		culture = { has_building_gfx = chinese_building_gfx }
		culture = { has_building_gfx = indian_building_gfx }
	}
}

artifact_use_indian_unit_visuals_trigger = {
	exists = culture
	OR = {
		culture = { has_unit_gfx = indian_unit_gfx }
		culture = { has_unit_gfx = austronesian_unit_gfx }
	}
}

artifact_should_use_gfx_type_trigger = {
	trigger_if = { #If the creator is not found we use the owner. Often this would be if the creator is a dummy, which has no culture
		limit = {
			OR = {
				AND = {
					exists = scope:artifact.creator.culture
					scope:artifact.creator.culture = { has_building_gfx = chinese_building_gfx }
				}
				AND = {
					exists = culture
					culture = { has_building_gfx = chinese_building_gfx }
				}
			}
		}
		artifact_use_indian_building_visuals_trigger = yes
	}
	trigger_if = { #If the creator is not found we use the owner. Often this would be if the creator is a dummy, which has no culture
		limit = {
			OR = {
				AND = {
					exists = scope:artifact.creator.culture
					scope:artifact.creator.culture = { has_unit_gfx = austronesian_unit_gfx }
				}
				AND = {
					exists = culture
					culture = { has_unit_gfx = austronesian_unit_gfx }
				}
			}
		}
		artifact_use_indian_unit_visuals_trigger = yes
	}
	trigger_else_if = {
		limit = {
			exists = scope:artifact.creator.culture
		}
		scope:artifact.creator = {
			artifact_use_$TYPE$_visuals_trigger = yes
		}
	}
	trigger_else_if = { #If the creator is not found we use the owner. Often this would be if the creator is a dummy, which has no culture
		limit = { exists = culture }
		artifact_use_$TYPE$_visuals_trigger = yes
	}
	trigger_else = {
		always = yes #We should never get here as all created artifacts should have an owner.
	}
}
