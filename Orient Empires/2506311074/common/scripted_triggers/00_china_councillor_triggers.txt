can_be_councillor_basics_trigger = {
	is_capable_adult = yes
	is_imprisoned = no
	is_at_war_with_liege = no	
	NOT = {
		has_realm_law = {
			temporal_head_of_faith_succession_law
		}
	}
}

can_be_neishijian_trigger = {
	can_be_councillor_basics_trigger = yes
	NOT = { has_council_position = councillor_spouse }
	has_trait = eunuch
	is_landed = no
	OR = {
		AND = {
			is_female = yes
			$COURT_OWNER$.faith = { NOT = { has_doctrine_parameter = male_dominated_council } }
		}
		AND = {
			is_male = yes
			$COURT_OWNER$.faith = { NOT = { has_doctrine_parameter = female_dominated_council } }
		}
	}
}

can_be_dowager_trigger = {
	save_temporary_scope_as = empress_dowager
	trigger_if = {
		limit = {
			$COURT_OWNER$ = { exists = father.father }
			$COURT_OWNER$.father.father = { exists = primary_spouse }
			$COURT_OWNER$.father.father.primary_spouse = { is_alive = yes }
			$COURT_OWNER$.father.father.primary_spouse = { is_courtier_of = $COURT_OWNER$ }
			$COURT_OWNER$.father.father.primary_spouse = { is_imprisoned = no }
		}
		$COURT_OWNER$.father.father.primary_spouse = scope:empress_dowager
	}	
	trigger_else_if = {
		limit = {
			$COURT_OWNER$ = { exists = father }
			$COURT_OWNER$.father = { exists = primary_spouse }
			$COURT_OWNER$.father.primary_spouse = { is_alive = yes }
			$COURT_OWNER$.father.primary_spouse = { is_courtier_of = $COURT_OWNER$ }
			$COURT_OWNER$.father.primary_spouse = { is_imprisoned = no }
		}
		$COURT_OWNER$.father.primary_spouse = scope:empress_dowager
	}
	trigger_else_if = {
		limit = {
			$COURT_OWNER$ = { exists = mother }
			$COURT_OWNER$.mother = { is_alive = yes }
			$COURT_OWNER$.mother = { is_courtier_of = $COURT_OWNER$ }
			$COURT_OWNER$.mother = { is_imprisoned = no }
		}
		$COURT_OWNER$.mother = scope:empress_dowager
	}
	trigger_else = {
		root.liege_or_court_owner.primary_spouse = scope:empress_dowager
	}	
}
