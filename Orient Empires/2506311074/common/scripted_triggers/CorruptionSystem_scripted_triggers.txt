CorruptionSystem_use_corruption_trigger = {
	liege = {
#		is_ai = no
		OR = {
			has_government = imperial_government
			has_government = corps_government
			has_government = bureau_government
		}
	}
}

	OR = {
		has_government = imperial_government
		has_government = corps_government
		has_government = bureau_government
	}	
}

	always = no
}
