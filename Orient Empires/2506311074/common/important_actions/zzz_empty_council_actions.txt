action_empty_council_position = {
	type = alert
	
	check_create_action = {
		if = {
			limit = {
				highest_held_title_tier > tier_barony
				has_free_council_slot = yes
				OR = {
					not = {
						any_councillor = {
							has_council_position = councillor_chancellor
						}
					}
					not = {
						any_councillor = {
							has_council_position = councillor_steward
						}
					}					
					not = {
						any_councillor = {
							has_council_position = councillor_spymaster
						}
					}
					not = {
						any_councillor = {
							has_council_position = councillor_marshal
						}
					}
					not = {
						any_councillor = {
							has_council_position = councillor_court_chaplain
						}
					}
				}
			}		
			try_create_important_action = {
				important_action_type = action_empty_council_position
				actor = root
			}	
		}
	}
	effect = {
		scope:actor = {
			open_view_data = {
				view = council_window
			}
		}
	}
}

action_powerful_vassal_not_on_council = {
	combine_into_one = yes	
	priority = 700
	
	check_create_action = {
		every_vassal = {
			limit = {
				is_powerful_vassal = yes
				is_councillor = no
				can_be_councillor_basics_trigger = yes
			}
			try_create_important_action = {
				important_action_type = action_powerful_vassal_not_on_council
				actor = root
				recipient = this
			}
		}
	}
	effect = {
		scope:actor = {
			open_view_data = {
				view = council_window
			}
		}
		scope:recipient = {
			open_view_data = {
				view = character
			}
		}
	}
}
