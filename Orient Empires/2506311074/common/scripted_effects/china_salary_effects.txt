china_got_council_position_effect = {

	if = {
		limit = {
			root.liege = {
				has_government = imperial_government
				has_title = title:e_celestia
			}
			highest_held_title_tier < tier_kingdom 
			is_bureau_government_trigger = yes
			highest_held_title_tier > tier_county
		}
		save_temporary_scope_as = temp_vassal
		liege = { save_temporary_scope_as = temp_liege }
		depose = yes
		scope:temp_liege = { 
			recruit_courtier = scope:temp_vassal
			send_interface_message = {
				type = event_generic_neutral	
				title = china_got_council_position_effect.tt #"职位候选人被夺去头衔，加入了你的宫廷"
				#desc = china_got_council_position_effect.desc #"职位候选人被夺去头衔，加入了你的宫廷"
				left_icon = scope:temp_vassal
			}
		}
		
	}
}	

china_got_marshal_position_effect = {

	if = {
		limit = {
			root.liege = {
				has_government = imperial_government
				has_title = title:e_celestia
			}
			highest_held_title_tier < tier_kingdom 
			highest_held_title_tier >= tier_county
			current_date < 1000.1.1
		}
		save_temporary_scope_as = temp_vassal
		liege = { save_temporary_scope_as = temp_liege }
		create_title_and_vassal_change = {
			type = granted
			save_scope_as = change
			add_claim_on_loss = no
		}
		title:d_dianqian = {
			change_title_holder = {
				holder = scope:temp_vassal
				change = scope:change
			}			
		}
		resolve_title_and_vassal_change = scope:change
		add_claim = title:e_celestia
	}
}	


china_fired_marshal_position_effect = {

	if = {
		limit = {
			root.liege = {
				has_government = imperial_government
				has_title = title:e_celestia
			}
			highest_held_title_tier < tier_kingdom 
			highest_held_title_tier >= tier_county
		}
		remove_claim = title:e_celestia
	}
}
