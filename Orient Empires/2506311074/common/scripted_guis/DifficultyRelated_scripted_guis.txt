DifficultyRelated_type_A_01_law_0_check_box_scripted_gui = {
	scope = character
	is_shown = { has_realm_law = DifficultyRelated_type_A_01_law_0 }
	effect = {
		if = {
			limit = { NOT = { has_realm_law = DifficultyRelated_type_A_01_law_0 } }
			add_realm_law = DifficultyRelated_type_A_01_law_0
		}
		else = { add_realm_law = DifficultyRelated_type_A_01_law_0 }
	}
}
