is_china_government = {
	scope = character

	is_shown = {
		is_landed = yes
		highest_held_title_tier = tier_empire
		OR = {
			is_oe_imperial_government_trigger = yes
			has_government = imperial_government
			has_government = ulus_government
		}
	}
}
is_china_government_or_vassal = {
	scope = character

	is_shown = {
		is_landed = yes
		highest_held_title_tier >= tier_duchy
		OR = {			
			has_government = imperial_government
			#has_government = ulus_government
			any_liege_or_above = {
				OR = {
					is_oe_imperial_government_trigger = yes
					has_government = imperial_government
				}
			}
		}
	}
}
