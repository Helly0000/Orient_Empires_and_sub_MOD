is_china_government = {
	scope = character
	is_shown = {	
		exists = this
		is_landed = yes
		is_oe_imperial_government_trigger = yes
	}
}
is_china_government_or_vassal = {
	scope = character
	is_shown = {
		exists = this
		is_landed = yes
		OR = {
			is_oe_imperial_official_trigger = yes
			top_liege = {
				is_oe_imperial_official_trigger = yes
			}
		}
	}
}
