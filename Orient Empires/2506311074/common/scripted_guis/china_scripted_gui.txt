is_china_government = { #东方内阁界面
	scope = character
	is_shown = {	
		exists = this
		is_landed = yes
		is_oe_imperial_official_trigger = yes
	}
}
is_china_government_or_vassal = { #东方朝廷界面
	scope = character
	is_shown = {
		exists = this
		is_landed = yes
		OR = {
			is_oe_imperial_official_trigger = yes
			liege = {
				is_oe_imperial_government_trigger = yes
			}
		}
	}
}

is_bureaucrat = { 
	scope = character
	is_shown = {
		is_oe_landed_bureaucrat_trigger = yes
	}
}
