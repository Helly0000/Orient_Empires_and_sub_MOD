is_china_government = {
	scope = character

	is_shown = {
		is_landed = yes
		has_government = imperial_government
		highest_held_title_tier = tier_empire
	}
}
is_china_government_or_vassal = {
	scope = character

	is_shown = {
		is_ai = no
		highest_held_title_tier >= tier_duchy
		OR = {
			has_government = imperial_government
			any_liege_or_above = {
				has_government = imperial_government
			}
		}
	}
}
