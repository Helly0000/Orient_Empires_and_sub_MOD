is_china_goverment = {
	scope = character

	is_shown = {
		exists = this
		is_landed = yes
		has_government = imperial_government
		highest_held_title_tier = tier_empire
	}
}
is_china_goverment_or_vassal = {
	scope = character

	is_shown = {
		exists = this
		is_landed = yes
		OR = {
			has_government = imperial_government
			any_liege_or_above = {
				has_government = imperial_government
			}
		}
	}
}
