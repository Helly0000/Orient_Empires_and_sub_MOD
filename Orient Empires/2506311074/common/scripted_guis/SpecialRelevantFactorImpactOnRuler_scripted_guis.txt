SpecialRelevantFactorImpactOnRuler_suggested_tributary_targets_update_list = {
	scope = character
	effect = {
		save_scope_as = gui_affected_ruler
		clear_variable_list = SpecialRelevantFactorImpactOnRuler_suggested_tributary_targets_list

		if = {
			limit = { SpecialRelevantFactorImpactOnRuler_trigger_ruler_is_type_A_01_trigger = yes }
			every_neighboring_top_liege_realm_owner = {
				limit = { SpecialRelevantFactorImpactOnRuler_relevant_factor_is_type_A_01_trigger = { RULER = gui_affected_ruler } }
				save_scope_as = gui_relevant_factor
				scope:gui_affected_ruler = {
					add_to_variable_list = {
						name = SpecialRelevantFactorImpactOnRuler_suggested_tributary_targets_list
						target = scope:gui_relevant_factor
					}
				}
			}		
		}

		if = {
			limit = { SpecialRelevantFactorImpactOnRuler_trigger_ruler_is_type_A_02_trigger = yes }
			every_neighboring_top_liege_realm_owner = {
				limit = { SpecialRelevantFactorImpactOnRuler_relevant_factor_is_type_A_02_trigger = { RULER = gui_affected_ruler } }
				save_scope_as = gui_relevant_factor
				scope:gui_affected_ruler = {
					add_to_variable_list = {
						name = SpecialRelevantFactorImpactOnRuler_suggested_tributary_targets_list
						target = scope:gui_relevant_factor
					}
				}
			}		
		}
	}
}
