CorruptionSystem_view_corruption_value_interaction = {

	category = interaction_category_vassal
	special_ai_interaction = recruit_courtier
	desc = CorruptionSystem.1001.tt

	is_shown = {
		always = no
		#scope:recipient = {
			#CorruptionSystem_use_corruption_trigger = yes
			#is_vassal_or_below_of = scope:actor
		#}
	}

	is_valid_showing_failures_only = {
		always = yes
	}	

	send_option = {
		is_valid = { always = yes }
		flag = Corruption_1
		localization = CorruptionSystem_view_corruption_value_interaction
	}
	send_option = {
		is_valid = { always = yes }
		flag = Corruption_2
		localization = CorruptionSystem.1002.tt
	}

	on_auto_accept = {
	}

	on_accept = {
		if = {
			limit = { scope:Corruption_1 = yes }
			scope:actor = {
				trigger_event = CorruptionSystem.1001
			}
			scope:recipient = { custom_tooltip = CorruptionSystem_view_corruption_value_interaction_desc }
		}
		if = {
			limit = { scope:Corruption_2 = yes }
			scope:actor = {
 			  trigger_event = CorruptionSystem.1002
				}
			scope:recipient = { custom_tooltip = CorruptionSystem_view_corruption_and_obligation_related_value_simple_interaction_desc }
		}

	}

	auto_accept = yes

	ai_potential = {
		always = no
	}
}
