common_join_faction_blockers = {
	# Do not join a new faction if I am already at war.
	modifier = {
		add = -1000
		AND = {
			is_at_war = yes
			trigger_if = {
				limit = {
					exists = joined_faction
				}
				NOT = { joined_faction = scope:faction }
			}	
		}
	}

	modifier = {
		add = -1000
		save_temporary_opinion_value_as = { name = target_opinion target = $FACTION_TARGET$ }
		scope:target_opinion >= 80
	}

	modifier = {
		add = -25
		has_trait = lifestyle_bureaucrat
	}
	modifier = {
		add = -25
		is_oe_landed_bureaucrat_trigger = yes
	}
	modifier = {
		add = 50
		is_legion_government_trigger = yes
	}
	modifier = {
		add = 25
		is_milita_government_trigger = yes
	}
	modifier = {
		add = 50
		vassal_contract_has_flag = imperial_royal_1
	}
}
