pool_bureaucracy_succession = { #这就是科举制
	valid_character = {
		is_capable_adult = yes
		is_imprisoned = no

		OR = {
			culture = { has_cultural_pillar = heritage_chinese }
			culture = { has_innovation = innovation_centralization_of_authority }
			has_same_culture_as = scope:base
			AND = {
				exists = scope:base.primary_title.title_province.culture
				culture = scope:base.primary_title.title_province.culture
			}
		}
		OR = {
			location.province_owner = { top_liege = scope:base.top_liege }
			location = scope:base.top_liege.capital_province
		}
		trigger_if = {
			limit = {
				NOT = { scope:base = { has_realm_law = female_only_law } }
			}
			is_male = yes
		}
		NOR = {
			has_trait = lunatic #纯质
			has_trait = possessed #中邪
			has_trait = peasant_leader #大逆
			has_trait = incapable #失能
			is_illegal_criminal_trigger = yes #众所周知的罪行
		}
	}
	
	character_score = {
		base = 0

		modifier = {
			has_trait_rank = {
				trait = BureaucracyEmpire_lifestyle_bureaucrat
				rank < 3
			}
			add = 20
		}
		modifier = { #随从
			add = 10
			is_courtier_of = scope:base
		}
		modifier = { #亲戚
			add = 10
			AND = {
				exists = dynasty
				exists = scope:base.dynasty
				dynasty = scope:base.dynasty				
			}
		}
		modifier = { #年级
			add = { value = age max = 30 }
		}
		modifier = { #教育
			add = 10
			has_education_learning_trigger = yes
		}
		#modifier = { #寒门
			#add = 10 	
			#is_lowborn = yes
		#}
		modifier = { #中产
			add = 10 	
			exists = dynasty
			dynasty = {
				NOT = { this = scope:base.top_liege.dynasty }
				any_dynasty_member = {
					is_alive = yes
					count > 3
				}
			}
		}
		modifier = { #门第
			dynasty_prestige_level > 0
			add = 10
			factor = {
				value = scope:base.dynasty.dynasty_prestige_level
			} 
		}
		#世家侧重
		modifier = {
			dynasty = { dynasty_prestige_level > 3 }
			add = -30
		}
		modifier = {
			dynasty = { dynasty_prestige_level > 5 }
			add = -20
		}
		modifier = { #考分
			add = learning
		}
		modifier = { #女子无才便是德
			factor = 0
			is_female = yes
		}
	}
	
	config = {
		background = bureaucrat
		age = { 16 55 }
	}
	
	selection_count = 10  #候补梯队
	
	pool = heritage
	court = no
	gender = male
}



pool_bureaucrat = { #这就是流官制
	valid_character = {
		is_capable_adult = yes
		is_imprisoned = no

		OR = {
			culture = { has_cultural_pillar = heritage_chinese }
			culture = { has_innovation = innovation_centralization_of_authority }
			has_same_culture_as = scope:base
			has_same_culture_as = scope:base.top_liege
			AND = {
				exists = scope:base.primary_title.title_province.culture
				culture = scope:base.primary_title.title_province.culture
			}
		}
		trigger_if = {
			limit = {
				NOT = { scope:base = { has_realm_law = female_only_law } }
			}
			is_male = yes
		}
		OR = {
			location.province_owner = { top_liege = scope:base.top_liege }
			location = scope:base.top_liege.capital_province
		}
		NOR = {
			has_trait = lunatic #纯质
			has_trait = possessed #中邪
			has_trait = peasant_leader #大逆
			has_trait = incapable #失能
			is_illegal_criminal_trigger = yes #众所周知的罪行
		}
	}
	
	character_score = {
		base = 0
		modifier = {
			add = 30
			OR = {
				location.province_owner = { top_liege = scope:base.top_liege }
				location = scope:base.top_liege.capital_province
			}
		}
		#资历
		modifier = {
			add = { value = bureaucracy_exp_value }
		}
		modifier = {
			add = { value = lifestyle_perk_points }
		}
		#幕僚
		modifier = {
			add = 10
			is_courtier_of = scope:base
		}
		#考分
		modifier = {
			add = learning
		}
		#门第
		modifier = {
			#dynasty.dynasty_prestige_level > 2
			dynasty_prestige_level > 0
			add = 20
			factor = dynasty.dynasty_prestige_level
		}
		modifier = {
			add = 10
			NOT = { dynasty = scope:base.top_liege.dynasty }
			any_close_or_extended_family_member = {
				is_alive = yes
				OR = {
					is_courtier_of = scope:base.top_liege
					target_is_liege_or_above = scope:base.top_liege
				}
			}
		}
		modifier = { #中产
			add = 10 	
			exists = dynasty
			dynasty = {
				NOT = { this = scope:base.top_liege.dynasty }
				any_dynasty_member = {
					is_alive = yes
					count > 3
				}
			}
		}
		modifier = {
			add = 20
			has_trait = jinshi
		}
		#官阶侧重
		modifier = {
			scope:base.primary_title.tier = tier_barony
			has_trait_rank = {
				trait = BureaucracyEmpire_lifestyle_bureaucrat
				rank = 1
			}
			add = 10
		}
		modifier = {
			scope:base.primary_title.tier = tier_county
			has_trait_rank = {
				trait = BureaucracyEmpire_lifestyle_bureaucrat
				rank < 3
			}
			add = 30
		}
		modifier = {
			scope:base.primary_title.tier = tier_duchy
			has_trait_rank = {
				trait = BureaucracyEmpire_lifestyle_bureaucrat
				rank >= 3
			}
			add = 30
		}
		#世家侧重
		modifier = {
			scope:base.primary_title.tier = tier_county
			dynasty = { dynasty_prestige_level < 4 }
			add = -40
		}
		modifier = {
			scope:base.primary_title.tier = tier_duchy
			dynasty = { dynasty_prestige_level < 5 }
			add = -50
		}
		modifier = { #女子无才便是德
			factor = 0
			is_female = yes
		}

		#}
