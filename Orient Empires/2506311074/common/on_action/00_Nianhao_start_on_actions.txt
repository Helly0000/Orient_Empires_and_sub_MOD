on_game_start_after_lobby = {  #by Maekawa
	on_actions = {
		nianhao_on_action
	}
}

nianhao_on_action = {
	on_actions = {
		Nianhao_867_start_effects #咸通
		Nianhao_882_start_effects #中和&金统
	}
}

Nianhao_867_start_effects = {	
	trigger = {
		current_date <= 867.1.1
	}
	effect = {
		character:longxi_li.1036 = {
			if = {
				limit = {
					is_alive = yes
					is_landed = yes
				}
				add_character_flag = nianhao_81
				set_variable = {
					name = nianhao_date
					value = 8
				}
			}
		}
	}
}

				}
