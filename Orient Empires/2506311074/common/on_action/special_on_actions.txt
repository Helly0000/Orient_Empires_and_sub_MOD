#藩镇继承、行政损耗、朝贡、军功等机制

on_game_start_after_lobby = {  
	on_actions = {
		corruption_effects
		tribute_effects
		celestia_throne_effects
	}
}

#逐级行政损耗
corruption_effects = {
	effect = {
		every_ruler = {	
			if = {
				limit = {
					top_liege = { has_primary_title = title:e_celestia }	
					is_independent_ruler = no
					has_government = imperial_government
					scope:title.tier <= tier_kingdom
				}
				modifier = {
					tax_factor = -0.3
					tax_mult = shire_corrupt_value
				}
			}
		}
	}
}

#朝贡加成
tribute_effects = {
	desc = tribute_effects
	effect = {
			every_character_to_title_neighboring_empire = {
				if = {
					limit = {
						has_culture = culture:han
						is_oe_imperial_government_trigger = yes
					}
				}
				development_growth_factor = 0.5
			}
		every_ruler = {			
			if = {
				limit = {
					NOT = { has_government = imperial_government }
					scope:title.tier <= tier_kingdom
					any_ally = { has_government = imperial_government }
				}
				modifier = {
					monthly_income_mult = 1
					monthly_prestige_gain_mult = -0.05
					monthly_dynasty_prestige_mult = 0.05
					domain_tax_mult = 0.15
					levy_reinforcement_rate = 0.15
					monthly_county_control_change_factor = 0.15
					root.culture = {
						add_random_innovation = yes
					}
				}
			}
		}
	}
}

#藩镇继承法
on_death = {
	on_actions = { 
		corps_succession_effect
	}
}


corps_succession_effect = {
	trigger = {
				has_government = imperial_government
				top_liege = { has_title = title:e_celestia }
				exists = capital_province #WLTDJFIX往来天地间修正
				capital_province = {
					geographical_region = world_asia_china
				}
				OR = {
					is_corps_government_trigger = yes
					has_realm_law = corps_succession_law
#					scope:title = { has_variable = corps_title } #WLTDJFIX往来天地间修正
				}
	}
	effect = {
		if = {
			limit = {
				exists = primary_heir
			}
			primary_heir = {
#WLTDJFIX往来天地间修正（开始）
				if = {
					limit = {
						is_independent_ruler = yes 
					}
					save_scope_as = liege
				}
				else_if = {
					limit = { exists = top_liege }
					top_liege = { save_scope_as = liege } 
				}
				else = {
					liege = { save_scope_as = liege }
				}
#WLTDJFIX往来天地间修正（开始）
				trigger_event = {
					days = 10
					id = bureaucracy.0001
				}
			}
		}
		else = {
			BureaucracyEmpire_bureaucracy_success_effect = {
				TITLE = root.primary_title #TITLE是被继任的头衔
				TEMP_HOLDER = root.primary_heir #TEMP_HOLDER是目前临时持有头衔的人，一般是死去的官员的孩子。用于正常交接
			}	
		}
	}
}


on_title_destroyed = {
	on_actions = { 
		fengjue_destroyed_effect
	}
}

fengjue_destroyed_effect = {
	trigger = {
		scope:landed_title = title:e_celestia
	}
	immediate = {
		if = {
			limit = {
				exists = scope:title:d_confucius
			}
			destroy_title = title:d_confucius
		}
	}
}
