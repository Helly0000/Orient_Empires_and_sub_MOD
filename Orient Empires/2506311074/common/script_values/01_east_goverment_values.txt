bureaucracy_exp_value_levelup = {
	value = 0
	if = {
		limit = { bureaucracy_exp_value < 7 }
		value = 7
		subtract = bureaucracy_exp_value
	}
	else_if = {
		limit = { bureaucracy_exp_value < 21 }
		value = 21
		subtract = bureaucracy_exp_value
	}
	else_if = {
		limit = { bureaucracy_exp_value < 42 }
		value = 42
		subtract = bureaucracy_exp_value
	}
	else_if = {
		limit = { bureaucracy_exp_value < 70 }
		value = 70
		subtract = bureaucracy_exp_value
	}
	else_if = {
		limit = { bureaucracy_exp_value < 105 }
		value = 105
		subtract = bureaucracy_exp_value
	}
	else = {
		value = 0
	}
}
