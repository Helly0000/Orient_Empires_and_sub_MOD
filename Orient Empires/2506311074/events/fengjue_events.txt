namespace = fengjue
#给予自定义绰号
fengjue.1 = {
	type = character_event
	title = fengjue.1.t
	desc = fengjue.1.desc
	theme = crown
	left_portrait = {
		character = root
	}
	right_portrait = {
		character = root
	}
	immediate = {	
	}

	widget = {
		gui = "event_window_widget_enter_nickname"
		container = "dynamic_birth_name"
		controller = {
			type = text
			data = {
				key = fengjue
				default = {
					localization_key = fengjue_wei
				}
				default = {
					localization_key = fengjue_shu
				}
				default = {
					localization_key = fengjue_wu
				}
			}
		}
		setup_scope = {
			root = { save_scope_as = text_target }
		}
	}
	option = { 
		name = newnianhao.a
	}
	after = {	
		if = {
			limit = { NOT = { has_variable = fengjue_var } } 
			set_variable = {
				name = fengjue_var
				value = flag:custom
			}
		}
		create_dynamic_title = {
			tier = kingdom
			name = fengjue_var_text
		}
		create_title_and_vassal_change = {
			type = created
			save_scope_as = change
			add_claim_on_loss = no
		}		
		scope:new_title = {
			generate_coa = yes 
			change_title_holder = {
				holder = root
				change = scope:change
			}
		}
		resolve_title_and_vassal_change = scope:change
	}
}

	option = {
		name = guohao.002.title
		trigger = {
			exists = scope:target
		}
		scope:recipient = {
			save_scope_as = holder
		}
		scope:actor = {
			hidden_effect = {
				if = {
					limit = {
						scope:kingdom = yes
					}
					add_prestige = 500
				}
				else = {
					add_prestige = 200
				}
				add_piety = {

			}
			trigger_event = fengjue.3
		}
	}

	option = {
		name = cancel

		scope:actor = {
			hidden_effect = {
				if = {
					limit = {
						scope:kingdom = yes
					}
					add_prestige = 500
				}
				else = {
					add_prestige = 200
				}
				add_piety = {
					value = 200
				}
			}
		}
	}

	after = {
		scope:recipient = { remove_character_flag = titular_holder }
		
	}

}
